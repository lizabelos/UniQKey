131  static inline void exportOperatingSystemKeyboards() {

135  dir.mkpath(".");
136  }

143  qDebug() << "Exporting keyboard" << keyboards[i];
144  QFile file("keyboards/" + keyboards[i] + ".keyboard");
145  file.open(QIODevice::WriteOnly);

157  static inline QList<QString> listExportedKeyboards(QString path=":/") {
