172  inline void serialize(QFile &file) const {

194  static inline Key deserialize(QFile &file) {
195  Key key;
196 
197  file.read((char*)(&key.mType), sizeof(KeyType));
198  file.read((char*)(&key.mXSpan), sizeof(float));
199  file.read((char*)(&key.mYSpan), sizeof(float));
200  file.read((char*)(&key.mX), sizeof(float));
201  file.read((char*)(&key.mY), sizeof(float));
202 
203  int mCharactersSize;
204  file.read((char*)(&mCharactersSize), sizeof(int));
205  if (mCharactersSize > 0) {
206  char* mCharactersBuffer = new char[mCharactersSize];
207  file.read(mCharactersBuffer, mCharactersSize);
208  key.mCharacters = QString(mCharactersBuffer);
209  delete[] mCharactersBuffer;
210  }
211 
212  return key;
213  }
