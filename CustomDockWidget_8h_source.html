54  void setCustomWidget(QWidget *widget) {
55  mContentWidget = new QWidget(this);
56  mContentLayout = new QVBoxLayout;
57  mContentLayout->setContentsMargins(0, 40, 0, 0); // Provide enough space for the title bar.
58  mContentWidget->setLayout(mContentLayout);
59  setWidget(mContentWidget);
60  mContentLayout->addWidget(widget);
61  }

63  protected:
64  void mousePressEvent(QMouseEvent *event) override {
65  // start dragging if mouse pressed on the title bar
66  if (event->button() == Qt::LeftButton && titleBarRect().contains(event->pos())) {
67  mDragStartPosition = event->pos();
68  mDrag = true;
69  if (!isFloating()) {
70  setFloating(true);
71  }
72  }
73 
74  if (event->button() == Qt::RightButton) {
75  setFloating(false);
76  }
77 
78  // start resizing if mouse pressed on the resize handle
79  if (event->button() == Qt::LeftButton && resizeHandleRect().contains(event->pos())) {
80  mResizeStartPosition = event->globalPos();
81  mResize = true;
82  }
83  }
84 
85  void mouseMoveEvent(QMouseEvent *event) override {
86  // move the widget
87  if (mDrag) {
88  QPoint diff = event->pos() - mDragStartPosition;
89  move(pos() + diff);
90  }
91 
92  // resize the widget
93  if (mResize) {
94  QPoint diff = mResizeStartPosition - event->globalPos();
95  QSize newSize = mContentWidget->size() + QSize(diff.x(), diff.y());
96  mContentWidget->setFixedHeight(newSize.height());
97  mContentWidget->setFixedWidth(newSize.width());
98  mResizeStartPosition = event->globalPos();
99 
100  // also move the widget if it is floating, to follow the mouse cursor
101  if (isFloating()) {
102  move(pos() - diff);
103  }
104  }
105  }
106 
107  void mouseReleaseEvent(QMouseEvent *event) override {
108  (void) event;
109  mDrag = false;
110  mResize = false;
111  }

121  // use event() to support qt5
122  bool event(QEvent *event) override {
123  if (event->type() == QEvent::TouchBegin && static_cast<QTouchEvent *>(event)->touchPoints().size() == 2) {
124  setFloating(false);
125  }
126  return QDockWidget::event(event);
127  }

129  void paintEvent(QPaintEvent *) override {
130  QPainter painter(this);
131 
132  // draw title bar
133  QRect titleBar = titleBarRect();
134  painter.fillRect(titleBar, Qt::lightGray);
135 
136  // draw resize handle
137  QRect resizeHandle = resizeHandleRect();
138  painter.drawText(resizeHandle, Qt::AlignCenter, "⤢");
139  }

150  QRect titleBarRect() const {
151  return QRect(resizeHandleRect().width(), 0, QWidget::width() - resizeHandleRect().width(), 40);
